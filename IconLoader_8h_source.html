203 if (const auto& icon = mIcons.find(name); icon != mIcons.end()) {
204 return icon->second.Texture;
205 }
206 if (const auto& icon = mLoadQueue.find(name); icon != mLoadQueue.end()) {
207 std::unique_lock lock(mLoadQueueMutex);

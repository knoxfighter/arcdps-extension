1034  // Handle resizing request
1035  // (We process this at the first TableBegin of the frame)
1036  // FIXME-TABLE: Contains columns if our work area doesn't allow for scrolling?
1037  if (mTable.InstanceCurrent == 0) {
1038  if (mTable.ResizedColumn != -1 && mTable.ResizedColumnNextWidth != FLT_MAX)
1039  SetColumnWidth(mTable.ResizedColumn, mTable.ResizedColumnNextWidth);
1040  mTable.LastResizedColumn = mTable.ResizedColumn;
1041  mTable.ResizedColumnNextWidth = FLT_MAX;
1042  mTable.ResizedColumn = -1;
1043 
1044  // Process auto-fit for single column, which is a special case for stretch columns and fixed columns with FixedSame policy.
1045  // FIXME-TABLE: Would be nice to redistribute available stretch space accordingly to other weights, instead of giving it all to siblings.
1046  if (mTable.AutoFitSingleColumn != -1) {
1047  SetColumnWidth(mTable.AutoFitSingleColumn, mTable.Columns[mTable.AutoFitSingleColumn].WidthAuto);
1048  mTable.AutoFitSingleColumn = -1;
1049  }
1050  }
1051 
1052  // Handle reordering request
1053  // Note: we don't clear ReorderColumn after handling the request.
1054  if (mTable.InstanceCurrent == 0) {
1055  if (mTable.HeldHeaderColumn == -1 && mTable.ReorderColumn != -1)
1056  mTable.ReorderColumn = -1;
1057  mTable.HeldHeaderColumn = -1;
1058  if (mTable.ReorderColumn != -1 && mTable.ReorderColumnDir != 0) {
1059  // We need to handle reordering across hidden columns.
1060  // In the configuration below, moving C to the right of E will lead to:
1061  // ... C [D] E ---> ... [D] E C (Column name/index)
1062  // ... 2 3 4 ... 2 3 4 (Display order)
1063  const int reorder_dir = mTable.ReorderColumnDir;
1064  IM_ASSERT(reorder_dir == -1 || reorder_dir == +1);
1065  IM_ASSERT(mTable.Flags & ImGuiTableFlags_Reorderable);
1066  TableColumn* src_column = &mTable.Columns[mTable.ReorderColumn];
1067  TableColumn* dst_column = &mTable.Columns[(reorder_dir == -1) ? src_column->PrevEnabledColumn : src_column->NextEnabledColumn];
1068  IM_UNUSED(dst_column);
1069  const int src_order = src_column->DisplayOrder;
1070  const int dst_order = dst_column->DisplayOrder;
1071  src_column->DisplayOrder = (TableColumnIdx) dst_order;
1072  for (int order_n = src_order + reorder_dir; order_n != dst_order + reorder_dir; order_n += reorder_dir)
1073  mTable.Columns[mTable.DisplayOrderToIndex[order_n]].DisplayOrder -= (TableColumnIdx) reorder_dir;
1074  IM_ASSERT(dst_column->DisplayOrder == dst_order - reorder_dir);
1075 
1076  // Display order is stored in both columns->IndexDisplayOrder and table->DisplayOrder[],
1077  // rebuild the later from the former.
1078  for (int column_n = 0; column_n < mTable.ColumnsCount; column_n++)
1079  mTable.DisplayOrderToIndex[mTable.Columns[column_n].DisplayOrder] = (TableColumnIdx) column_n;
1080  mTable.ReorderColumnDir = 0;
1081  mTable.IsSettingsDirty = true;
1082  }
1083  }
1084

1086  if (mTable.IsResetDisplayOrderRequest) {
1087  for (int n = 0; n < mTable.ColumnsCount; n++)
1088  mTable.DisplayOrderToIndex[n] = mTable.Columns[n].DisplayOrder = (TableColumnIdx) n;
1089  mTable.IsResetDisplayOrderRequest = false;
1090  mTable.IsSettingsDirty = true;
1091  }

2540  // - All fixed: easy.
2541  // - All stretch: easy.
2542  // - One or more fixed + one stretch: easy.
2543  // - One or more fixed + more than one stretch: tricky.

2549  // - F1 F2 F3 resize from F1| or F2| --> ok: alter ->WidthRequested of Fixed column. Subsequent columns will be offset.
2550  // - F1 F2 F3 resize from F3| --> ok: alter ->WidthRequested of Fixed column. If active, ScrollX extent can be altered.

2552  // - F1 F2 W3 resize from W3| --> ok: no-op (disabled by Resize Rule 1)
2553  // - W1 W2 W3 resize from W1| or W2| --> ok
2554  // - W1 W2 W3 resize from W3| --> ok: no-op (disabled by Resize Rule 1)
2555  // - W1 F2 F3 resize from F3| --> ok: no-op (disabled by Resize Rule 1)
2556  // - W1 F2 resize from F2| --> ok: no-op (disabled by Resize Rule 1)
2557  // - W1 W2 F3 resize from W1| or W2| --> ok
2558  // - W1 F2 W3 resize from W1| or F2| --> ok
2559  // - F1 W2 F3 resize from W2| --> ok
2560  // - F1 W3 F2 resize from W3| --> ok
2561  // - W1 F2 F3 resize from W1| --> ok: equivalent to resizing |F2. F3 will not move.
2562  // - W1 F2 F3 resize from F2| --> ok
2563  // All resizes from a Wx columns are locking other columns.

2565  // Possible improvements:
2566  // - W1 W2 W3 resize W1| --> to not be stuck, both W2 and W3 would stretch down. Seems possible to fix. Would be most beneficial to simplify resize of all-weighted columns.
2567  // - W3 F1 F2 resize W3| --> to not be stuck past F1|, both F1 and F2 would need to stretch down, which would be lossy or ambiguous. Seems hard to fix.
2568
